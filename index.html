    <script>

        let currentEnergy = 0;


        function completeTask(btn, energy) {

            if (btn.disabled) return;

            

            // 基礎互動：禁用按鈕、震動

            btn.disabled = true;

            btn.style.opacity = "0.5";

            if (navigator.vibrate) navigator.vibrate(200);


            // 更新能量

            currentEnergy += energy;

            document.getElementById('progress-bar').style.width = currentEnergy + "%";


            // 對話更新

            const texts = ["太棒了！光回來了！", "謝謝你，森林在笑呢！", "黑霧正在散去！"];

            // 檢查是否達成修復

            if (currentEnergy >= 100) {

                document.body.classList.add('restored');

                document.getElementById('lumi-text').innerText = "世界恢復了！你是最棒的守護者！";

                // 這裡可以加入慶祝音效

            }

        }
